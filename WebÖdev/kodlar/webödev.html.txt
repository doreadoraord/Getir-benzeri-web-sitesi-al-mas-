                const phoneInput = document.getElementById('phoneInput');

                phoneInput.value = "0";
            
                phoneInput.addEventListener('input', function (e) {
                    // Sadece sayılar yazılabilsin
                    this.value = this.value.replace(/[^0-9]/g, '');
            
                    // 11 karakteri geçmesin
                    if (this.value.length > 11) {
                        this.value = this.value.slice(0, 11);

                    // Başına otomatik "0" koy

<script>
    const loginModal = document.getElementById("loginModal");
    const kayitModal = document.getElementById("kayitModal");
    const loginBtn = document.getElementById("loginBtn");
    const kayitBtn = document.getElementById("kayitBtn");
    const closeButtons = document.querySelectorAll(".close");

    loginBtn.onclick = function() {
        loginModal.style.display = "block";
    };

    kayitBtn.onclick = function() {
        kayitModal.style.display = "block";
    };

    closeButtons.forEach(closeBtn => {
        closeBtn.onclick = function() {
            closeBtn.closest('.modal').style.display = "none";
        };
    });

    window.onclick = function(event) {
        if (event.target == loginModal || event.target == kayitModal) {
            loginModal.style.display = "none";
            kayitModal.style.display = "none";
        }
    };
</script>
